# Python Unit Tests Rules

Unit tests in the notebooks repository validate Python scripts, utilities, and build tools without requiring container execution.

## When to Write Unit Tests

Write unit tests when you need to:
- Test Python build scripts (e.g., `scripts/pylocks_generator.py`)
- Test utility functions and helpers
- Validate parsers, formatters, or data transformers
- Test configuration loaders
- Test URL manipulation or string processing
- Test logic that doesn't require Docker/container execution
- Test file system operations with `tmp_path`

## Framework and Tools

| Tool | Purpose |
|------|---------|
| **pytest** | Test framework |
| **pytest fixtures** | Test setup and dependency injection |
| **tmp_path** | Temporary directory fixture |
| **pathlib** | Path manipulation |

## Test File Structure

### File Naming and Location

**Pattern**: `test_*.py` or `*_test.py`

**Location**: `tests/unit/` with subdirectories matching the code structure

Example:
- Script: `scripts/pylocks_generator.py`
- Test: `tests/unit/scripts/test_pylocks_generator.py`

### Test Organization

```python
"""Unit tests for module_name."""

from __future__ import annotations

from pathlib import Path

# Import module under test

    """Test description."""

class TestClassName:
    """Tests for ClassName functionality."""
    
    def test_method_name(self) -> None:
        """Test description."""
```

## Unit Test Patterns

### 1. Basic Function Test

```python
def test_utc_now_iso_format() -> None:
    """Test that UTC timestamp is formatted correctly."""
    s = pg.utc_now_iso()
    assert len(s) == 20
    assert s.endswith("Z")
    assert "T" in s
```

**Key Points**:
- One test per logical behavior
- Multiple assertions are OK if testing one concept
- Descriptive test name describes what is being tested
- Docstring provides additional context

### 2. Parametrized Tests with Test Classes

```python
class TestEnsureJsonFormatParam:
    """Tests for ensure_json_format_param function."""
    
    def test_bare_url(self) -> None:
        assert pg.ensure_json_format_param("https://example.com/simple/") == (
            "https://example.com/simple/?format=json"
        )
    
    def test_no_trailing_slash(self) -> None:
        assert pg.ensure_json_format_param("https://example.com/simple") == (
            "https://example.com/simple?format=json"
        )
    
    def test_already_has_format_json(self) -> None:
        url = "https://example.com/simple/?format=json"
        assert pg.ensure_json_format_param(url) == url
    
    def test_overwrites_non_json_format(self) -> None:
        assert pg.ensure_json_format_param("https://example.com/simple/?format=html") == (
            "https://example.com/simple/?format=json"
        )
    
    def test_other_query_params(self) -> None:
        assert pg.ensure_json_format_param("https://example.com/simple/?other=1") == (
            "https://example.com/simple/?other=1&format=json"
        )
```

**Key Points**:
- Group related tests in a class
- Class name starts with `Test`
- Each test method covers one scenario
- Clear test names describe the scenario
- Use multi-line formatting for long expected values

### 3. Tests with tmp_path Fixture

```python
def test_parse_exclude_newer_equals_form(tmp_path: Path) -> None:
    """Test parsing --exclude-newer=VALUE format from lockfile."""
    p = tmp_path / "pylock.toml"
    p.write_text(
        "# This file was autogenerated by uv via the following command:\n"
        "#    uv pip compile pyproject.toml --exclude-newer=2025-06-01T12:30:45Z --quiet\n"
        'lock-version = "1.0"\n',
        encoding="utf-8",
    )
    assert pg.parse_exclude_newer_from_lockfile(p) == "2025-06-01T12:30:45Z"
```

**Key Points**:
- Use `tmp_path` fixture for temporary files
- `tmp_path` is a `pathlib.Path` to a unique temporary directory
- Directory is automatically cleaned up after test
- Always specify `encoding="utf-8"` for text files
- Use `/` operator for path joining

### 4. Tests with Custom Fixtures

```python
@pytest.fixture
def repo_root() -> Path:
    """Return the repository root path."""
    return Path(__file__).resolve().parents[3]

def test_parse_exclude_newer_real_cpu_lock_header(repo_root: Path) -> None:
    """Test parsing from actual lockfile in repository."""
    sample = (
        repo_root / "jupyter" / "datascience" / "ubi9-python-3.12" / 
        "uv.lock.d" / "pylock.cpu.toml"
    )
    if not sample.is_file():
        pytest.skip("sample lockfile not present")
    parsed = pg.parse_exclude_newer_from_lockfile(sample)
    assert parsed is not None
```

**Key Points**:
- Define fixtures with `@pytest.fixture` decorator
- Fixtures can return any type
- Type hint fixture return value
- Use `pytest.skip()` when test dependencies are missing
- Test against real files when appropriate

### 5. Edge Case and Error Handling Tests

```python
def test_parse_exclude_newer_missing(tmp_path: Path) -> None:
    """Test that missing --exclude-newer returns None."""
    p = tmp_path / "pylock.toml"
    p.write_text(
        "# This file was autogenerated by uv via the following command:\n"
        "#    uv pip compile pyproject.toml --quiet\n"
        'lock-version = "1.0"\n',
        encoding="utf-8",
    )
    assert pg.parse_exclude_newer_from_lockfile(p) is None

def test_parse_exclude_newer_missing_file(tmp_path: Path) -> None:
    """Test that non-existent file returns None."""
    assert pg.parse_exclude_newer_from_lockfile(tmp_path / "nope.toml") is None
```

**Key Points**:
- Test both success and failure paths
- Test missing files, empty input, None values
- Test boundary conditions
- Use `assert result is None` for None checks
- Descriptive names for edge cases

### 6. Multiple Parameter Variations

```python
def test_parse_exclude_newer_equals_form(tmp_path: Path) -> None:
    """Test --exclude-newer=VALUE format."""
    p = tmp_path / "pylock.toml"
    p.write_text(
        "# ... --exclude-newer=2025-06-01T12:30:45Z ...\n",
        encoding="utf-8",
    )
    assert pg.parse_exclude_newer_from_lockfile(p) == "2025-06-01T12:30:45Z"

def test_parse_exclude_newer_space_form(tmp_path: Path) -> None:
    """Test --exclude-newer VALUE format (space separated)."""
    p = tmp_path / "pylock.toml"
    p.write_text(
        "# ... --exclude-newer 2025-06-01T12:30:45Z ...\n",
        encoding="utf-8",
    )
    assert pg.parse_exclude_newer_from_lockfile(p) == "2025-06-01T12:30:45Z"

def test_parse_exclude_newer_offset_form(tmp_path: Path) -> None:
    """Test timestamp with timezone offset."""
    p = tmp_path / "pylock.toml"
    p.write_text(
        "# ... --exclude-newer=2025-06-01T12:30:45+00:00 ...\n",
        encoding="utf-8",
    )
    assert pg.parse_exclude_newer_from_lockfile(p) == "2025-06-01T12:30:45+00:00"
```

**Key Points**:
- Test different input formats
- Each variation gets its own test
- Clear docstrings distinguish variations
- Consistent assertion patterns

### 7. Testing with Boolean Flags

```python
def test_resolve_exclude_newer_live_mode(tmp_path: Path) -> None:
    """Test live mode uses provided timestamp."""
    p = tmp_path / "pylock.toml"
    p.write_text("# no exclude\n", encoding="utf-8")
    assert pg.resolve_exclude_newer(
        p,
        ci_check=False,
        live_timestamp="2020-01-01T00:00:00Z"
    ) == "2020-01-01T00:00:00Z"

def test_resolve_exclude_newer_ci_parsed(tmp_path: Path) -> None:
    """Test CI mode parses from file."""
    p = tmp_path / "pylock.toml"
    p.write_text(
        "# ... --exclude-newer=2099-01-01T00:00:00Z\n",
        encoding="utf-8",
    )
    assert pg.resolve_exclude_newer(
        p,
        ci_check=True,
        live_timestamp="2020-01-01T00:00:00Z"
    ) == "2099-01-01T00:00:00Z"
```

**Key Points**:
- Test with different boolean flag combinations
- Use keyword arguments for clarity
- Test names describe the mode/configuration

## Module Import Patterns

### Importing Module Under Test

```python
"""Unit tests for pylocks_generator exclude-newer helpers."""

from __future__ import annotations

from pathlib import Path
import pytest

_REPO_ROOT = Path(__file__).resolve().parents[3]

import scripts.pylocks_generator as pg  # noqa: E402
```

**Key Points**:
- Import module under test after path setup if needed
- Use short aliases (`pg`, `mod`) for frequently used modules
- Add `# noqa: E402` if importing after code (path manipulation)
- Use `from __future__ import annotations` for forward references

### Alternative Import Pattern

```python
from __future__ import annotations

from pathlib import Path
import pytest

# Import directly if module is in PYTHONPATH
from scripts import pylocks_generator
```

## Assertion Patterns

### String Assertions

```python
assert s == "expected"
assert s.startswith("prefix")
assert s.endswith("suffix")
assert "substring" in s
assert len(s) == 20
```

### None Assertions

```python
assert result is None
assert result is not None
```

### Equality Assertions

```python
assert result == expected
assert result == ("value1", "value2")  # tuples
assert result == {"key": "value"}      # dicts
```

### Collection Assertions

```python
assert len(items) == 5
assert item in collection
assert set(items) == expected_set
```

## Best Practices Summary

### DO ✅
- Write one test per logical behavior
- Use descriptive test names
- Add docstrings to tests
- Use `tmp_path` for file operations
- Test both success and failure paths
- Test edge cases and boundary conditions
- Use type hints for fixtures
- Use `pytest.skip()` for conditional tests
- Group related tests in classes
- Use multi-line formatting for readability
- Specify `encoding="utf-8"` for text files
- Use `pathlib.Path` for path operations

### DON'T ❌
- Don't test implementation details
- Don't duplicate test logic
- Don't use `assert True` or meaningless assertions
- Don't leave commented-out tests
- Don't create files outside `tmp_path`
- Don't use `sleep()` or timing-based tests
- Don't test external services (mock them)
- Don't use mutable default arguments in fixtures
- Don't hardcode absolute paths
- Don't test third-party library behavior

- [ ] List edge cases to cover
- [ ] Check for existing similar tests

- [ ] Import module under test

## Running Unit Tests

```bash
# Run all unit tests
pytest tests/unit/

# Run specific test file
pytest tests/unit/scripts/test_pylocks_generator.py

# Run specific test
pytest tests/unit/scripts/test_pylocks_generator.py::test_utc_now_iso_format

# Run with verbose output
pytest -v tests/unit/

# Run with coverage
pytest --cov=scripts tests/unit/
```

## Common Fixtures

### Built-in pytest Fixtures

| Fixture | Type | Purpose |
|---------|------|---------|
| `tmp_path` | `pathlib.Path` | Unique temporary directory per test |
| `tmp_path_factory` | `TempPathFactory` | Session-scoped temporary directory factory |
| `capsys` | `CaptureFixture` | Capture stdout/stderr |
| `monkeypatch` | `MonkeyPatch` | Mock/patch objects |

### Using tmp_path

```python
def test_file_creation(tmp_path: Path) -> None:
    # Create file in temporary directory
    file = tmp_path / "test.txt"
    file.write_text("content", encoding="utf-8")
    
    # Read and validate
    assert file.read_text(encoding="utf-8") == "content"
    
    # tmp_path is automatically cleaned up after test
```

### Using capsys

```python
def test_print_output(capsys) -> None:
    print("Hello, World!")
    captured = capsys.readouterr()
    assert captured.out == "Hello, World!\n"
    assert captured.err == ""
```

## Example: Complete Test File

```python
"""Unit tests for example_module utilities."""

from __future__ import annotations

from pathlib import Path
import pytest

import scripts.example_module as em

@pytest.fixture
def sample_config(tmp_path: Path) -> Path:
    """Create sample configuration file."""
    config = tmp_path / "config.toml"
    config.write_text(
        'key = "value"\n'
        'number = 42\n',
        encoding="utf-8",
    )
    return config

class TestConfigParser:
    """Tests for configuration parsing."""
    
    def test_parse_valid_config(self, sample_config: Path) -> None:
        """Test parsing valid configuration file."""
        result = em.parse_config(sample_config)
        assert result["key"] == "value"
        assert result["number"] == 42
    
    def test_parse_missing_file(self, tmp_path: Path) -> None:
        """Test parsing non-existent file returns None."""
        result = em.parse_config(tmp_path / "missing.toml")
        assert result is None
    
    def test_parse_invalid_toml(self, tmp_path: Path) -> None:
        """Test parsing invalid TOML raises error."""
        invalid = tmp_path / "invalid.toml"
        invalid.write_text("invalid {{{ toml", encoding="utf-8")
        
        with pytest.raises(ValueError):
            em.parse_config(invalid)

def test_format_timestamp() -> None:
    """Test timestamp formatting."""
    timestamp = em.format_timestamp("2025-01-01T00:00:00Z")
    assert timestamp.endswith("Z")
    assert len(timestamp) == 20
```
